    Messages = function(container, mini, onChange) {
        // Display message
        this.error = function(message, heading, context) {
            this._message('as-error', message, (heading ? heading : ''), context);
        };
        this.warn = function(message, heading, context) {
            this._message('as-warning', message, (mini || heading ? heading : ''), context);
        };
        this.success = function(message, heading, context) {
            this._message('as-success', message, (mini || heading ? heading : ''), context);
        };
        this.message = function(message, heading, context) {
            this._message('as-message', message, (mini || heading ? heading : ''), context);
        };

        // Hide messages
        this.hide = function(context) {
            container.find('div.as-message'+(context ? '.'+context : '')).remove();
            if (onChange) onChange();
        };

        // Private
        this._message = function(messageClass, message, heading, context) {
            this.hide();

            $('<div class="hidden as-message ' + messageClass + (mini ? ' as-mini-message' : '') + ' ' + (context ? context : '') + '">' +
              '  <span class="messageIcon'  + (mini ? ' miniIcon ' : '') + '"></span>' +
              (heading ? '  <h6>' + heading + '</h6>' : '') +
              '  <p>'  + message + '</p>' +
              '</div>').appendTo(container).fadeIn();

            if (onChange) onChange();
        };
    };
})(jQuery)});

  function AmazonLikeStateCache(stateCacheId) {
      var self = this;

      self.init   = function () {};
      self.ready  = function () {};
      self.get    = function () {};
      self.set    = function () {};

      self.getStateWhenReady = function (callback) {
          self.ready(function(){
              callback(
                  (self.get('isLiked') == "true"),
                  parseInt(self.get('customerWhitelistStatus')),
                  parseInt(self.get('likeCount')),
                  self.get('commifiedLikeCount'),
                  self.get('commifiedLikeCountMinusOne')
              );
          });
      };

      self.getTimestamp = function () {
          var ts = self.get('ts');
          if (ts != null && typeof(ts) != 'undefined' && ts.length > 0) {
              ts = parseInt(ts);
              if (!isNaN(ts) && ts > 0) {
                 return ts;
              }
          }
          return 0;
      };

      var ephemeralCache = {};

      // Override behavior based on browser capabilities and context we are in
      if (typeof(sessionStorage) != "undefined") {
          self.ready = function (onReadyCallback) {
              onReadyCallback();
          };
          self.get = function (stateName) {
              return sessionStorage.getItem(stateCacheId + "_" + stateName);
          };
          self.set = function (stateName, stateValue) {
              sessionStorage.setItem(stateCacheId + "_" + stateName, stateValue + "");
          };
      } else {
          // If this is DP, fallback to ephemeral (page-level) caching only
          self.init = function () {
             ephemeralCache[stateCacheId] = {};
          };
          self.ready = function (onReadyCallback) {
              onReadyCallback();
          };
          self.get = function (stateName) {
              return ephemeralCache[stateCacheId][stateName];
          };
          self.set = function (stateName, stateValue) {
              ephemeralCache[stateCacheId][stateName] = stateValue + "";
          };
        }
        return this;
    };

            } else {
                amznLikePopover.likePopoverUI.find('.likeOffPopoverContent').show();
                amznLikePopover.likePopoverUI.find('.likePopoverError').show();
            }
            
            amznLikePopover.fixupAmzPopover();
        };
        
        this.onUnlike = function (success) {
            amznLikePopover.clearContent();
            amznLikePopover.onLikeCountUpdate();
            
            if (success) {
                amznLikePopover.likePopoverUI.find('.likeOffSavedPopoverContent').show();

        this.likeContainer = likeContainer;

                "itemId"    : amznLike.itemId,
                "itemType"  : amznLike.itemType,
                "sessionId" : amznLike.sessionId
            };

                cache   : false,
                dataType: 'json',
                timeout : 5000,
                url     : url,
                data    : postData,

                    callback(response);
                },
                error   : function(xmlHttpRequest, textStatus, errorThrown) {
                    callback({});
                }
            });
        };

                    callback(response);
                },
                error   : function(xmlHttpRequest, textStatus, errorThrown) {
                    callback({});
                }
            });

        };

        // Attach to DOM
        amznLike.attachToDOM();

        return this;
    }

amznJQ.available("JQuery", function(){
(function($){
  if(typeof($.amz_taf_getTwisterHandlers) != "undefined") return; 
  // Get the selected variant.
  var getSelectedVariant = function(dpState){
    return dpState['selected_variation_values'];
  }
  // Get the dimenstions from the variations that are not -1 (- 1 means not selected).
  var getSelectedDimensions = function(variationValues){
    if(!variationValues) return null;
    var validKeys = [];
    var noOfDimensions = 0;
    for (var dimension in variationValues){
      noOfDimensions++;
      if(typeof(variationValues[dimension]) != "undefined" && variationValues[dimension] != -1){
        validKeys[validKeys.length] = dimension;
      }
    }
   // This is an optimization. If all the dimensions are selected, the feature is 
   // autmatically refreshed using sever side calls, so return without processing.
   if(validKeys.length == noOfDimensions) return null;
   return validKeys;
  } 

  // Get a ASIN from the variation. This method iterates over all the child ASINs and picks the first ASIN that 
  // matches all the selected dimensions.
  var selectedASIN = null;
  var getAsinFromVariant = function(asinVariations, selectedVariant, validKeys){
    if(!asinVariations || !selectedVariant || !validKeys){
      return null;
    }
    for(var obj in asinVariations){
      var valid = true;
      for(var j = 0 ; j < validKeys.length; j++ ){
        if(!asinVariations[obj][ validKeys[j]]){
          continue;
        }
        if(asinVariations[obj][ validKeys[j]] != selectedVariant[validKeys[j]]){
          valid = false;
          break;
        }
      }
      if(valid){
        selectedASIN = asinVariations[obj]["ASIN"];
        return true;
      }
    }
    return false;
  }
  // Check if a variation is already selected. This will happen for the case when you have 
  // selected a size and go to unselected.
  var isAlreadySelected = function(validKeys, selectedVariant, variations){
    if(!selectedASIN || !variations) return false;
    for(var i = 0; i< validKeys.length; i++){
      if(variations[selectedASIN][validKeys[i]] != selectedVariant[validKeys[i]]) return false;
    }
    return true;
  }
  var updateHrefs = function(elementId){
    var element = jQuery('#' + elementId);
    var href = jQuery(element).attr('href');
    if(href.indexOf("childASIN") >= 0 ){
      href = href.replace(/childASIN=.*/g, "childASIN=" + selectedASIN);
    }else{
      href += "&childASIN=" + selectedASIN;
    }
    // Set parent ASIN if available.
    // Read http://w.amazon.com/index.php/Traffic/SocialMedia/TAF/TwisterSupport on why this is done.
    if(parentASIN){
      if(href.indexOf("parentASIN") >= 0 ){
        href = href.replace(/parentASIN=[^\&]*/g, "parentASIN=" + parentASIN);
      }else{
        href += "&parentASIN=" + parentASIN;
      }
    }
    jQuery(element).attr('href', href);
  }
  var updateSocialNetworkLinks = function(elementId){
    var element = jQuery('#' + elementId);
    var socialButtons = jQuery(element).parent().find('.tafSocialButton').parent();
    for(var i = 0; i < socialButtons.length; i++){
      var socialButton = jQuery(socialButtons[i]);
      var href = socialButton.attr('href');
      href = href.replace(/%2Fdp%2F[^\/%]*%2Fref/g, "%2Fdp%2F" + selectedASIN + "%2Fref");
      var onClick = socialButton.attr('onClick');
      if(onClick && !onClick.replace){
        // IE does not expose onclick as an attribute through attr, so we use getAttributeNode to read the 
        // value and update the onClick event. We also set another property called oldOnClick on the first 
        // update of onClick event, this is used next time so that we can read the onclick as string.
        if(socialButton.length > 0 && socialButtons[i].getAttributeNode){
          if(socialButtons[i].getAttributeNode('oldOnClick')){
            onClick = socialButtons[i].getAttributeNode('oldOnClick').value; 
          }else if(socialButtons[i].getAttributeNode('onClick')){
            onClick = socialButtons[i].getAttributeNode('onClick').value; 
          }
          if(onClick && onClick.replace && onClick != "null"){ 
            onClick = onClick.replace(/%2Fdp%2F[^%\/]*%2Fref/g, "%2Fdp%2F" + selectedASIN + "%2Fref");
            socialButton.unbind('click');
            // Work around for IE which does not support setAttribute("onClick");
            socialButtons[i].setAttribute('oldOnClick', onClick);
            onClick = new Function(onClick);
            socialButtons[i].onclick = onClick;
          }
        }
      }else if(onClick){
        onClick = onClick.replace(/%2Fdp%2F[^%\/]*%2Fref/g, "%2Fdp%2F" + selectedASIN + "%2Fref");
        socialButton.unbind('click');
        socialButton.attr('onClick', onClick);
      }
      socialButton.attr('href', href);
    }
  }
  var updateEventsForShare = function(){
    for(triggerID in amz_taf_triggers){
      // Unbind the events for all the triggers and add back with the new child ASIN;
      if(amz_taf_triggers && amz_taf_triggers[triggerID] && amz_taf_triggers[triggerID].isCompact){
        jQuery('#' + triggerID).removeAmazonPopoverTrigger();
        amz_taf_bindCompactPopoverTrigger(triggerID);
        var hiddenDivID = "tafPopHidden_" + triggerID;
        jQuery('#' + hiddenDivID).removeAmazonPopoverTrigger();
        amz_taf_bindLightBoxTrigger(hiddenDivID, selectedASIN, parentASIN);
        updateHrefs("tafCompact_" + triggerID ); 
      }
      amz_taf_bindLightBoxTrigger(triggerID, selectedASIN, parentASIN);
      // Also update the href for the trigger ID appropriately
      updateHrefs(triggerID );
      //Update FB and Twitter URLS:
      // For twitter we have to update the url="" parameter and for FB we have to update u="" to the correct ASIN. 
      // This can be done using regex on the href and onClick;
      updateSocialNetworkLinks(triggerID);
    }
  }
  var asin_deselect = function(dpState){
    var selectedVariant = getSelectedVariant(dpState);
    var validKeys = getSelectedDimensions(selectedVariant);
    if(validKeys == null) return;
    if(isAlreadySelected(validKeys, selectedVariant, dpState['asin_variation_values'])){
      return;
    }
    if(!getAsinFromVariant(dpState['asin_variation_values'], selectedVariant, validKeys)){
      return;
    }
    if(dpState['parent_asin']){
      parentASIN = dpState['parent_asin'];
    }
    updateEventsForShare();
  }
  
  var document_ready = function(dpState){
    var selectedVariant = getSelectedVariant(dpState);
    var validKeys = getSelectedDimensions(selectedVariant);
    if(validKeys == null) return;
    if(isAlreadySelected(validKeys, selectedVariant, dpState['asin_variation_values'])){
      return;
    }
    if(!getAsinFromVariant(dpState['asin_variation_values'], selectedVariant, validKeys)){
      return;
    }
    updateEventsForShare();
  }
  $.amz_taf_getTwisterHandlers = function(event_type){
    if(event_type == "asin_deselect")  return asin_deselect;
    if(event_type == "document_ready")  return document_ready;
  }
  var twisterInitialized = false;
  $.amz_taf_initTwisterHandler = function (){
    if(twisterInitialized) return;
    amznJQ.available("twister", function(){
      if(!DetailPageFramework || !DetailPageFramework.registerCallback){
        return;
      }
      DetailPageFramework.registerFeatureConfig("_tell_a_friend", {suppressDefaultBehaviour:true, dataType: 'btf'});
  //    Not required. This was required for a case where the images changes based on availability, 
  //    but there is no guarantee that the document_ready event is not already fired. There is no other way to get the event.
  //    DetailPageFramework.registerCallback("document_ready", "_tell_a_friend", jQuery.amz_taf_getTwisterHandlers("document_ready"));
      DetailPageFramework.registerCallback("asin_deselect", "_tell_a_friend", jQuery.amz_taf_getTwisterHandlers("asin_deselect"));
    });
    twisterInitialized = true;
  }
})(jQuery);
});
function amz_taf_generatePopover(triggerID, useTwister) {
  var hashValue = window.location.hash;
  var hashIndex; 
  var swfHashValue = "_" + triggerID + "_Swf"; 
  var isRedirect = false;
  hashIndex = hashValue.indexOf(swfHashValue);
  if (hashValue != "" && hashIndex > -1) { 
    isRedirect = true;
  }
  var url = window.location.pathname ;
  url += window.location.search; 
  if(window.location.hash && window.location.hash != "" && window.location.hash != "#"){
    var hash = window.location.hash.replace(/_[^&?/]*_Swf/, "");
    url += hash;
  }

  if(!isRedirect){
    if (url.indexOf("#") > -1){
      url += "&" + swfHashValue;
    }else{
      url += "#" + swfHashValue;
    }
  }

  if(encodeURIComponent){
    url=encodeURIComponent(url);
  }else if(encodeURI){
    url=encodeURI(url);
  }else{
    url = escape(url); 
  }
  amz_taf_triggers[triggerID].url = amz_taf_triggers[triggerID].url +   "&redirectLocation=" + url;
  amz_taf_triggers[triggerID].origTriggerID = triggerID;
  var isCompact = amz_taf_triggers[triggerID].isCompact;
  var hiddenDivID = "tafPopHidden_" + triggerID;
  if(isCompact && amz_taf_triggers[hiddenDivID]){
    amz_taf_triggers[hiddenDivID].url = amz_taf_triggers[hiddenDivID].url +   "&redirectLocation=" + url;
    amz_taf_triggers[hiddenDivID].origTriggerID = triggerID;
    jQuery("#" + triggerID).after( '<div id="' + hiddenDivID + '" style="display:none"></div>');
    // Bind the compact popover
    amz_taf_bindCompactPopoverTrigger(triggerID);
    amz_taf_bindLightBoxTrigger(hiddenDivID);
  }
  amz_taf_bindLightBoxTrigger(triggerID);
  if(isRedirect && !amz_taf_showSwfCalled){

/*
 * Update the popover for SWF.
 * Args:
 * triggerID: The ID for the trigger link (don't prepend ID with "#")
 * newParams: A hash of parameters to pass to change for the popover.
 *   All of the current parameters that are not specified within the hash
 *   will be unchanged.
 *   FIXME: This current implementation only can change params that
 *   were originally specified on creation of the SWF link
 */
function amz_taf_updatePopoverTarget(triggerID, newParams){
  var url = amz_taf_triggers[triggerID].url;
  if(newParams) {
    url += '&';
    jQuery.each(newParams, function(key, val) {
      url = url.replace(new RegExp(key+"=(:?.*?)&"), key+"="+escape(val)+'&');
    });
    amz_taf_triggers[triggerID].url = url.substr(0, url.length-1);
  }

  jQuery("#" + triggerID).removeAmazonPopoverTrigger();
  amz_taf_generatePopover(triggerID);
}

/*
 * Client-side validation of e-mail addresses. Returns an array of invalid
 * e-mail addresses.
 */
function amz_taf_validateRecipients(src) {
  var recipients = jQuery.trim(src.value);
  recipients = recipients.split(/[,;\s]+/);

  var unique = {};
  for (var i in recipients) {
    unique[recipients[i]] = 1;
  }

  var invalid = new Array();
  for (var i=0; i< unique.length; i++) {
    var e = unique[i]; 
    if (!e.match(/((?:[\w\.+-]*?[a-zA-Z0-9+])?@(?:[a-zA-Z0-9][\w-]*\.)+[a-zA-Z]{2,4})(.*)/)) {
      invalid.push(e);
    }
  }

  return invalid;
}

function amz_taf_validateMessage(src) {
  var message = src.value;

  //this is to remove all '\r's so IE accurately counts the characters
  var re = /[\r]/g;
  message=message.replace(re,"");

  var maxLength = amz_taf_config.maxLength;
  var messageLength = message.length;
  var charsLeft = maxLength - messageLength ;

  var rulesObj = document.getElementById("taf_send_message_rules");
  var rulesErrorObj = document.getElementById("taf_send_message_rules_error");

  // abort if not defined
  if (!rulesObj || !rulesErrorObj)
    return true;

  // these change depending on whether we want to show an error or not
  var charsLeftObj = "taf_msg_charsLeft";
  var pluralObj = "taf_msg_plural";

  var isError = charsLeft < 0;

  if (!isError) {
    rulesObj.style.display = '';
    rulesErrorObj.style.display = 'none';

  } else {
    rulesObj.style.display = 'none';
    rulesErrorObj.style.display = '';
    charsLeftObj += "_error";
    pluralObj += "_error";

    // adjust for display
    charsLeft = -charsLeft;
  }

  charsLeftObj = document.getElementById(charsLeftObj);
  pluralObj = document.getElementById(pluralObj);

  //abort if not defined:
  if (!charsLeftObj) {
     return !isError;
  }

  charsLeftObj.innerHTML = charsLeft;
  if(!pluralObj) return !isError; 
  pluralObj.innerHTML = (charsLeft == 1) ? '' : 's';

  return !isError;
}

function amz_taf_sendButton_onSubmit(form) {
    if (amz_taf_popover.processing){
        return false;
    } else {
        amz_taf_popover.processing = 1;
    }
    
    //  put the number of email recipients in the reftag  
    var friendsArr = document.getElementsByName('friendsRecipients');

    var recipientsObj = form.recipients;

    if ( !friendsArr || !recipientsObj ) {
        return false;
    }

    var postContent = {};

    var count = 0;
    var recipients = recipientsObj.value;
    if ( recipients != amz_taf_config.detailedInstructions ) {
        // Separating email addresses by comma+space. TT:0011087001
        recipients = recipients.replace(/[,;]\s*/g, ", ");

        // Trimming
        recipients = recipients.replace(/^(,|;|\s)+/,"");
        recipients = recipients.replace(/(,|;|\s)+$/,"");

        if ( recipients != "" ) {
            postContent.recipients = recipients;
        }
    }

    var selectedFriends = [];
    for ( var i = 0; i < friendsArr.length; i++ ) {
        if ( friendsArr[i].checked ) {
            selectedFriends.push(friendsArr[i].value);
            count++;
        }
    }
    if (selectedFriends.length > 0){
        postContent.friendsRecipients = selectedFriends;
    }
    
    if(form.message && form.message.value.length > 0 && form.message.value != amz_taf_config.messageInstructions){
        postContent.message=form.message.value;
    }
    if (form.copyMeBox.checked){
        postContent.copyMeBox="on";
    }
    if (form.inviteBox && form.inviteBox.checked){
        postContent.inviteBox="on";
    }

    var fields = {
      ces: 1,
      contentID: 1,
      contentName: 1,
      contentType: 1,
      contentURI: 1,
      emailTemplate: 1,
      guess: 1,
      imageURL: 1,
      params: 1,
      placementID: 1,
      titleText: 1,
      channel: 1,
      eventID: 1,
      params: 1,
      merchantID: 1,
      viaAccount: 1,
      relatedAccounts: 1,
      parentASIN: 1,
      isDynamicSWF: 1,
      emailSubjectStrID: 1,
      emailCustomMsgStrID: 1,
      learnMoreButton: 1,
      emailCaptionStrID: 1,
      emailDescStrID: 1,
     
/*
* These fields are  generated by $Std->getFormInternatinoalizationFields() method and are used by gurupa to correctly decode form fields
* https://w.amazon.com/index.php/Gurupa/UserDocs/CharacterEncoding#Overview
*/
      __mk_ja_JP: 1,
      __mk_de_DE: 1,
      __mk_fr_FR: 1,
      __mk_zh_CN: 1,
      referer: 1
    };
    for (var field in fields) {
      if (fields.hasOwnProperty(field) && typeof form[field] !== 'undefined') {
        postContent[field] = form[field].value;
      }
    }

    postContent.token=form.token.value;

    jQuery.post(amz_taf_config.link(count), postContent, amz_taf_sendDoneCallback);

    return false;
}
